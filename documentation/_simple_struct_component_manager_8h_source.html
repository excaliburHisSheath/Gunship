131  void DestroyImmediate( Entity::ID entityID )
132  {
133  // Retrieve the index of the component to be destroyed, then
134  // remove the component to be destroyed from the index map.
135  size_t index = _componentIndices[entityID];
136  _componentIndices.erase( entityID );
137 
138  // If the component isn't at the end of the vector,
139  // swap the last component into the destroyed component's spot.
140  // Otherwise we can just pop the end of the vector.
141  if ( index != _components.size() - 1 )
142  {
143  // Since the old component isn't at the end of the vector,
144  // the component we're moving shouldn't have the same entityID.
145  SDL_assert_paranoid( _components.back().entityID != entityID );
146

149
